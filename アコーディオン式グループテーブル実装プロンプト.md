## ステップ2: グループ集計ロジックを追加

現在の `combinedProductRanking`, `productGroups`, `filteredProducts`, `groupTotal*` の変数群（104〜142行目）をすべて削除し、以下に置き換える:

```typescript
// Amazon + 楽天 統合の商品データ（グループ情報・チャネル付き）
const combinedProducts = [
  ...(productSummary as any[]).map((p: any) => ({
    ...p,
    channel: "Amazon" as const,
    group: p.product?.product_group || "その他",
  })),
  ...(rakutenProductSummary as any[]).map((p: any) => ({
    ...p,
    channel: "楽天" as const,
    group: p.product?.product_group || "その他",
  })),
];

// 商品ランキング TOP5 用（既存のランキングカードで使用）
const combinedProductRanking = combinedProducts
  .map(p => ({
    name: p.product?.name || "不明",
    channel: p.channel,
    group: p.group,
    total_sales: p.total_sales,
    total_orders: p.total_orders,
    net_profit: p.net_profit || 0,
  }))
  .sort((a, b) => b.total_sales - a.total_sales);

// グループ別に集計
interface GroupedData {
  groupName: string;
  children: any[];
  total_sales: number;
  total_orders: number;
  total_ad_spend: number;
  net_profit: number;
  profit_rate: number;
}

const groupedProductData: GroupedData[] = (() => {
  const groups: Record<string, GroupedData> = {};
  for (const p of combinedProducts) {
    const key = p.group;
    if (!groups[key]) {
      groups[key] = {
        groupName: key,
        children: [],
        total_sales: 0,
        total_orders: 0,
        total_ad_spend: 0,
        net_profit: 0,
        profit_rate: 0,
      };
    }
    groups[key].children.push(p);
    groups[key].total_sales += p.total_sales || 0;
    groups[key].total_orders += p.total_orders || 0;
    groups[key].total_ad_spend += p.total_ad_spend || 0;
    groups[key].net_profit += p.net_profit || 0;
  }
  // 利益率を計算
  for (const g of Object.values(groups)) {
    g.profit_rate = g.total_sales > 0 ? (g.net_profit / g.total_sales) * 100 : 0;
  }
  // ソート
  return Object.values(groups).sort((a, b) => {
    if (groupSortKey === "profit_rate") return b.profit_rate - a.profit_rate;
    if (groupSortKey === "net_profit") return b.net_profit - a.net_profit;
    if (groupSortKey === "total_orders") return b.total_orders - a.total_orders;
    return b.total_sales - a.total_sales; // デフォルト: 売上順
  });
})();

// グループ展開トグル
const toggleGroup = (key: string) => {
  setExpandedGroups(prev => {
    const next = new Set(prev);
    if (next.has(key)) next.delete(key);
    else next.add(key);
    return next;
  });
};

// 全体合計
const grandTotalSales = groupedProductData.reduce((sum, g) => sum + g.total_sales, 0);
const grandTotalOrders = groupedProductData.reduce((sum, g) => sum + g.total_orders, 0);
const grandTotalAdSpend = groupedProductData.reduce((sum, g) => sum + g.total_ad_spend, 0);
const grandTotalProfit = groupedProductData.reduce((sum, g) => sum + g.net_profit, 0);
const grandTotalProfitRate = grandTotalSales > 0 ? (grandTotalProfit / grandTotalSales) * 100 : 0;
```
